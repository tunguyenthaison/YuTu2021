\section{Interior cone condition}
\begin{defn} A domain $\Omega$ satisfies the \emph{interior cone condition} if 
\begin{itemize}
    \item[(A1)] for each $z\in \partial\Omega$, there exists positive numbers $r_0,s_0 \in (0,1)$ such that
    \begin{equation}\label{e:int_cone}
    \forall\;x\in \overline{\Omega}\cap B_{r_0}(z): \qquad B(x-r\,\mathbf{n}(z),rs_0) \subset \Omega \qquad\text{for all}\; r\in (0,r_0).
\end{equation}
\end{itemize}
\end{defn}
\noindent Comparing this with H. Soner's condition, which is stated as follows.
\begin{itemize}
    \item[(A2)] There exists a universal pair of positive numbers $(r,h)$ and $\eta\in \mathrm{BUC}(\overline{\Omega};\mathbb{R}^n)$ such that $B(x+t\eta(x), rt)\subset\Omega$ for all $t\in (0,h]$.
\end{itemize}
